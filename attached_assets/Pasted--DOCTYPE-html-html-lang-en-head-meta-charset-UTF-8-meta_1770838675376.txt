<script>
function showScreen(index) {
  // Remove active from all screens and nav buttons
  document.querySelectorAll('.screen-view').forEach(s => s.classList.remove('active'));
  document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
  
  // Show selected screen
  const screen = document.getElementById('screen-' + index);
  screen.classList.add('active');
  
  // Re-trigger animations
  const meta = screen.querySelector('.screen-meta');
  const phone = screen.querySelector('.phone-wrap');
  const hint = screen.querySelector('.flow-hint');
  
  if (meta) {
    meta.style.animation = 'none';
    meta.offsetHeight;
    meta.style.animation = 'fadeUp 0.5s ease forwards';
  }
  if (phone) {
    phone.style.animation = 'none';
    phone.offsetHeight;
    phone.style.animation = 'phoneIn 0.6s ease 0.15s forwards';
  }
  if (hint) {
    hint.style.animation = 'none';
    hint.offsetHeight;
    hint.style.animation = 'fadeUp 0.5s ease 0.4s forwards';
  }
  
  // Activate nav button
  document.querySelectorAll('.nav-btn')[index].classList.add('active');
  
  // Scroll to top
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

// Keyboard navigation
document.addEventListener('keydown', (e) => {
  const current = [...document.querySelectorAll('.screen-view')].findIndex(s => s.classList.contains('active'));
  if (e.key === 'ArrowRight' && current < 26) showScreen(current + 1);
  if (e.key === 'ArrowLeft' && current > 0) showScreen(current - 1);
});
</script>
